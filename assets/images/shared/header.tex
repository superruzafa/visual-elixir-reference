\newcommand\braceflow[4]{
	\draw [line width=0.4mm, decorate, decoration={brace, amplitude=2.5mm, raise=2.5mm}] (#1) -- (#2);
	\draw [line width=0.4mm, decorate, decoration={brace, amplitude=2.5mm, raise=2.5mm}] (#3) -- (#4);
	\coordinate (last first brace) at ($ ($ (#1)!.5!(#2) $) + (0, -5mm) $);
	\coordinate (last second brace) at ($ ($ (#3)!.5!(#4) $) + (0, 5mm) $);
	\draw [->] (last first brace) -- (last second brace);
}
