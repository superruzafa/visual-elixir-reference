\newcommand\applyfunabove[3]{
    \node (last value) [above=2 of #1] {#3};
    \node (last function) at ($ (#1)!.5!(last value) $) [function, in=south, out=north] {#2};
    \draw [->] (#1) -- (last function.in);
    \draw [->] (last function.out) -- (last value);
}
