\newcommand\applyfuntop[4][2]{
  \node (last label) [above=#1 of #2] {#4};
  \node (last function) at ($ (last label.base)!.5!(#2) $) [function in=south, function out=north, fun=#3];
  \draw [->] (#2) -- (last function.in);
  \draw [->, shorten >=1mm] (last function.out) -- (last label.base);
}

\newcommand\bracetobrace[4]{
  \draw [line width=.4mm, decorate, decoration={brace, raise=1mm, amplitude=2mm}] (#1) -- (#2);
  \draw [line width=.4mm, decorate, decoration={brace, raise=1mm, amplitude=2mm}] (#3) -- (#4);
  \coordinate (last brace start) at  ($ (#1)!.5!(#2) + (0, -3mm) $); 
  \coordinate (last brace end) at  ($ (#3)!.5!(#4) + (0, 3mm) $); 
  \draw [->] (last brace start) -- (last brace end);
}
