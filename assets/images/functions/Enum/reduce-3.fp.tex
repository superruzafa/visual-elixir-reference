	fun/.style={function, anchor=north port 1, north ports=2, node contents={\texttt{fun}}},
	fun1/.style={fun,},
	fun2/.style={fun,},
	funi/.style={fun,},
	funn/.style={fun},
}

\matrix [list=a] {
    \node [index=1]; &
    \node [index=2]; &
    \node [elements between]; &
    \node [index=i]; &
    \node [elements between]; &
	\node [index=n]; \\
};

\foreach \i in {1,2,i,n}{
	\draw [->] (a\i.south) -- +(0, -1)
		node (fun\i) [fun\i];
}
